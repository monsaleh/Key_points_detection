def load():
    X_nodrop = np.stack([format_img(string) for string in df['Image']]).astype(np.float32)[:, :, :, np.newaxis] #without_dropna
    
    XX = np.stack([format_img(string) for string in df_drop['Image']]).astype(np.float32)[:, :, :, np.newaxis]
    yy = np.vstack(df_drop[df_drop.columns[:-1]].values).astype(np.float32)
    
    X= XX/255. # image pixels from 0 to 1
    y= (yy-48)/48 # target values from -1 to 1
    
    return X_nodrop, X, y
